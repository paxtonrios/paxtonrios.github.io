需求:0~9之间可以组成多少不重复的三位数.

2022-11-26

代码如下:

#include"stdio.h"
main()
{
	int number,number1,number2,sum,l=0;
	for(number=1;number<10;number++)
		for(number1=0;number1<10;number1++)
		for(number2=0;number2<10;number2++)
	{
		sum=number*100+number1*10+number2;
		if(sum%100/10!=sum/100&&sum%100/10!=sum%10&&sum/100!=sum%10)
	{
		printf("%d ",sum);
		l++;
		if(l!=0)
		if(l%20==0)
		printf("\n");
	}
	}
		return 0;
}
